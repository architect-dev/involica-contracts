contract InvolicaTaskManager is OpsReady {
    using SafeERC20 for IERC20;
    using EnumerableSet for EnumerableSet.AddressSet;
    using EnumerableSet for EnumerableSet.UintSet;
 
    Involica public involica;
    address public constant NATIVE_TOKEN = 0xEeeeeEeeeEeEeeEeEeEeeEEEeeeeEeeeeeeeEEeE;
 
    receive() external payable {}
 
    constructor(
        address payable _involica,
        address _ops
    ) OpsReady(_ops) {
        involica = Involica(_involica);
    }
 
    // TASK MANAGEMENT
 
    function checkAndInitializeTask(IInvolica.Position memory _position) external returns (bool initialized, bytes32 taskId) {
        initialized = true;
        if (_position.user == address(0)) initialized = false;
        if (_position.taskId != bytes32(0)) initialized = false;
 
        if (initialized) {
            taskId = IOps(ops).createTimedTask(
                0,
                uint128(_position.intervalDCA),
                address(this),
                involica.executeDCA.selector,
                involica.resolver(),
                abi.encodeWithSelector(IInvolicaResolver(involica.resolver()).checkPositionExecutable.selector, _position.user),
                NATIVE_TOKEN,
                false
            );
        }
    }
 
    function checkShouldFinalizeTask(IInvolica.Position memory _position, uint256 _txFee) external view returns (bool finalize, string memory reason, bool emptyTreasury) {
        // Funds must be approved for this contract
        if (
            _position.tokenIn != address(0) &&
            IERC20(_position.tokenIn).allowance(_position.user, address(this)) < _position.amountDCA
        ) {
            return (true, 'Insufficient approval to pull from wallet', false);
        }
 
        // Must be enough funds for DCA
        if (
            _position.tokenIn != address(0) && IERC20(_position.tokenIn).balanceOf(_position.user) < _position.amountDCA
        ) {
            return (true, 'Insufficient funds to pull from wallet', false);
        }
 
        if (involica.userTreasuries(_position.user) == 0 || involica.userTreasuries(_position.user) < _txFee) {
            if (_txFee > 0) {
                // Tx has fee, but isn't covered by user's treasury, empty to zero
                emptyTreasury = true;
            }
 
            return (true, 'Treasury out of gas', emptyTreasury);
        }
 
        return (false, '', false);
    }
 
    function finalizeTask(IInvolica.Position memory _position) external returns (bool finalized) {
        if (_position.taskId == bytes32(0)) return false;
        IOps(ops).cancelTask(_position.taskId);
        return true;
    }
}
